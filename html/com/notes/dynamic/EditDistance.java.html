3    public class EditDistance { 
4     
5        // the x string 
6        String x; 
7     
8        // the y strings 
9        String y; 
10    
11       // integer array to store the minimum distance between x[1..i] and y[1..j] 
12       // editDist[i][j] = minimum distance between x[1..i] and y[1..sj] 
13       int[][] editDist; 
14       String[][] sols; 
15    
16       //Approach 
17       /** 
18        * editDist[i][j] =  editDist[i-1][j-1]+1 if x[i]= y[j] 
19        *                   min { editDist[i][j-1] + 1 convert x[1..i] into y[1..j-1] and insert y[j] 
20        *                         editDist[i-1][j] + 1 convert x[1..i-1] into y[1..j] and delete x[j] 
21        *                         editDist[i-1][j-1]+1 convert x[1..i-1] into y[1..j-1] and replace x[i] with y[j] 
22        *                       } 
23        * Do following initializations 
24        * editDist[i][0] = i; // for all i as we just need to delete x[i]; 
25        * editDist[0][j] = j; // for all j as we just need to insert y[j]; 
26        * editDist[0][0] = 0  // to convert string of length 0 to 0. 
27        * <p/> 
28        * Required solution will be in editDistance[x.length][y.length]; 
29        */ 
30       void calculateEditDist() { 
31           editDist = new int[x.length() + 1][y.length() + 1]; 
32           sols = new String[x.length() + 1][y.length() + 1]; 
33    
34           // edit distance between 2 string of length 0 is 0. 
35           editDist[0][0] = 0; 
36           sols[0][0] = ""; 
37           //editDist[i][0] = i; for all i as we just need to delete x[i]; 
38           initForXintoLengthZero(); 
39           //editDist[0][j] = j; // for all j as we just need to insert y[j]; 
40           intoForLenthZeroIntoY(); 
41           // loop to build the solution from bottom up 
42           for (int i = 1; i <= x.length(); i++) { // length of string x from 1 to x.length() 
43               for (int j = 1; j <= y.length(); j++) {// length of string y from 1 to y.length() 
44                   if (x.charAt(i - 1) == y.charAt(j - 1)) { // if current char is same 
45                       editDist[i][j] = editDist[i - 1][j - 1];  // number of operations are same as that of sub problem 
46                       sols[i][j] = sols[i - 1][j - 1]; // solution is same as that of sub problem 
47                   } else { 
48                       // number of operations required for converting string of length i into string of j-1 length and insert y[j] 
49                       int insert = editDist[i][j - 1] + 1; 
50                       // number of operations required for converting string of length i-1 into string of j length and delete x[i] 
51                       int delete = editDist[i - 1][j] + 1; 
52                       // number of operations required for converting string of length i-1 into string of j-1 length and replace x[i] with y[j] 
53                       int replace = editDist[i - 1][j - 1] + 1; 
54                       int min = Math.min(insert, Math.min(delete, replace)); 
55                       editDist[i][j] = min; 
56                       if (min == insert) { 
57                           sols[i][j] = sols[i][j - 1] + "Insert " + y.charAt(j - 1) + "\n"; 
58                       } else if (min == delete) { 
59                           sols[i][j] = sols[i - 1][j] + "Delete " + x.charAt(i - 1) + "\n"; 
60                       } else if (min == replace) { 
61                           sols[i][j] = sols[i - 1][j - 1] + "Replace " + x.charAt(i - 1) + " with " + y.charAt(j - 1) + "\n"; 
62                       } 
63                   } 
64               } 
65           } 
66           System.out.println("Edit Distance between " + x + " and " + y + " is " + editDist[x.length()][y.length()]); 
67           System.out.println("Steps are \n" + sols[x.length()][y.length()]); 
68       }

70       private void intoForLenthZeroIntoY() { 
71           for (int j = 1; j <= y.length(); j++) { 
72               // edit distance to convert a string of length 0 to string of length j is j. 
73               editDist[0][j] = j; 
74               String ops = ""; 
75               for (int k = 0; k < j; k++) { 
76                   // we need to perform j insert operations. 
77                   ops = ops + "Insert " + y.charAt(k) + "\n"; 
78               } 
79               sols[0][j] = ops; 
80           } 
81       } 
82    
83       private void initForXintoLengthZero() { 
84           for (int i = 1; i <= x.length(); i++) { 
85               // edit distance to convert a string of length i to string of length 0 is i. 
86               editDist[i][0] = i; 
87               String ops = ""; 
88               for (int k = 0; k < i; k++) { 
89                   // we need to perform i delete operations on x. 
90                   ops = ops + "Delete " + x.charAt(k) + "\n"; 
91               } 
92               sols[i][0] = ops; 
93           } 
94       }
