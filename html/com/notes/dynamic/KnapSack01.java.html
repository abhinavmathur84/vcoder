4    public class KnapSack01 { 
5     
6        int[] weight; 
7     
8        // benefit[i] corresponding to weight[i] 
9        int[] benefit; 
10    
11       // size of knapsack 
12       int size; 
13    
14       int[] B; 
15       String[] sols; 
16       //Approach 
17    
18       /** 
19        * B[w] = max benefit which can be obtained from knapsack of size w 
20        * B[w] = max{B[w-weight[i]]+benefit[i]} for all i possible in weights 
21        */ 
22       void calcuate() { 
23           B = new int[size + 1]; 
24           sols = new String[size + 1]; 
25    
26           B[0] = 0; // benefit for knapsack of size 0 is 0; 
27           sols[0] = ""; 
28    
29           int len = benefit.length; 
30    
31           // build the solution bottom up 
32           for (int w = 0; w <= size; w++) { 
33               int max = Integer.MIN_VALUE; 
34               for (int i = 0; i < len; i++) { 
35                   if (w >= weight[i]) { 
36                       int t = B[w - weight[i]] + benefit[i]; 
37                       if (t > max) { 
38                           max = t; 
39                           B[w] = t; 
40                           sols[w] = sols[w - weight[i]] + "," + weight[i]; 
41                       } 
42                   } 
43               } 
44           } 
45           System.out.println("Optimal Weight is " + B[size]); 
46           System.out.println("Solution is " + sols[size]); 
47       } 
48    
49       public static void main(String[] args) { 
50           int[] w = {2, 3, 4, 5}; 
51           int[] b = {3, 4, 5, 6}; 
52    
53           KnapSack01 ks = new KnapSack01(); 
54           ks.benefit = b; 
55           ks.weight = w; 
56           ks.size = 5; 
57           ks.calcuate(); 
58    
59           int val[] = {60, 100, 120}; 
60           int wt[] = {10, 20, 30}; 
61           int  W = 50; 
62           ks.benefit = val; 
63           ks.weight = wt; 
64           ks.size = W; 
65           ks.calcuate(); 
66       }
