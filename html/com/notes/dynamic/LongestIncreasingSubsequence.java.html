3    public class LongestIncreasingSubsequence { 
4     
5        // the array 
6        int[] a; 
7     
8        //lisLength[i]= Length of longest increasing subsequence ending at a[i]. 
9        int[] lisLength; 
10    
11       // sols[i] = longest increasing subsequence ending at a[i]; 
12       String[] sols; 
13    
14       //Approach 
15       /** 
16        * Initialize 
17        * lisLength[0] = 1 
18        * sols[0] ="a[0]"; 
19        * lisLength[i] = max{lisLength[j]}+1 for all 0 < j < i and a[i] > a[j] 
20        *              =  1 otherwise 
21        * sols[i] =  sols[j] + a[i] where sols[j] is correspondingly chosen 
22        *         =  a[i] otherwise 
23        * <p/> 
24        * We are required to compute max of lisLength and corresponding sols 
25        */ 
26       void calculateLIS() { 
27           lisLength = new int[a.length]; 
28           sols = new String[a.length]; 
29    
30           lisLength[0] = 1; // lis for array of length 1 ending at a[0]. 
31           sols[0] = a[0] + ""; 
32    
33           // required data we need to compute 
34           int maxLisLength = Integer.MIN_VALUE; 
35           String lis = ""; 
36           // build the solution bottom up 
37           for (int i = 1; i < a.length; i++) { 
38               int max = Integer.MIN_VALUE; 
39               String sol = ""; 
40               // loop over all the previous elements 
41               for (int j = 0; j < i; j++) { 
42                   if (a[j] < a[i]) { // checking the increasing subsequence property 
43                       if (max < lisLength[j]) { // check if current max is less than lis ending at a[j] 
44                           max = lisLength[j]; // if true then update the max value 
45                           sol = sols[j]; // update the solution 
46                       } 
47                   } 
48               } 
49              // no increasing subsequence found which end in a[i] 
50               if (max == Integer.MIN_VALUE) { 
51                   lisLength[i] = 1; 
52                   sols[i] = a[i] + ""; 
53               } else { 
54                   // now update lisLength[i] 
55                   lisLength[i] = max + 1; 
56                   sols[i] = sol + "," + a[i]; 
57               } 
58               // update the global max which we need 
59               if (maxLisLength < lisLength[i]) { 
60                   maxLisLength = lisLength[i]; 
61                   lis = sols[i]; 
62               } 
63           } 
64    
65           System.out.println("Length of longest increasing subsequence " + maxLisLength); 
66           System.out.println("Longest increasing subsequence is \n" + lis); 
67       }
