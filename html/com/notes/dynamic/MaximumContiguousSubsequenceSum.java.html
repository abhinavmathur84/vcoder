3    public class MaximumContiguousSubsequenceSum { 
4     
5        // The input array 
6        int[] a; 
7     
8        // array which contains the maximum sum till i; 
9        // maxSum[i] = maximum sum till a[i]; 
10       int[] maxSum; 
11    
12       // array which stores the subsequence which will give the max sum till i; 
13       // sols[i] = subsequence which will give maximum sum till a[i] 
14       String[] sols; 
15    
16       //Approach 
17    
18       /** 
19        * initialize the max  to maxSum[0]; 
20        * Solve for i-1 and now 
21        * if maxSum[i-1] > 0 then maxSum[i] = max {maxSum[i-1]+a[i],a[i]} and update the sols accordingly 
22        * else maxSum[i] = a[i] and sols[i] = a[i] only. start a new window at a[i] 
23        * keep updating the max and solution accordingly 
24        */ 
25    
26       void calculate() { 
27           maxSum = new int[a.length]; 
28           sols = new String[a.length]; 
29    
30           // initialize the maxSum with just 1 element array as that element itself; 
31           maxSum[0] = a[0]; 
32    
33           // initialize the solution(subsequence) with just 1 element array as that element itself.s 
34           sols[0] = a[0] + ""; 
35    
36           int max = maxSum[0]; 
37           String sol = sols[0]; 
38           // loop to build the subsequence bottom up starting from second element of array and ending at last. 
39           for (int i = 1; i < a.length; i++) { 
40               if(maxSum[i-1] > 0) { 
41                   int temp = maxSum[i-1]+a[i]; 
42                   if(temp > a[i]) { 
43                       maxSum[i] = temp; // update the maxSum 
44                       sols[i] = sols[i-1]+","+a[i]; // update the sols to include a[i] 
45                   } else { 
46                       maxSum[i] = a[i]; //update the maxSum to a[i] 
47                       sols[i] = a[i]+""; // start a new subsequence starting from a[i] 
48                   } 
49               } else { 
50                   maxSum[i] = a[i]; //update the maxSum to a[i] 
51                   sols[i] = a[i]+""; // start a new subsequence starting from a[i] 
52               } 
53               if(max < maxSum[i]) { 
54                   max = maxSum[i]; 
55                   sol = sols[i]; 
56               } 
57           } 
58           System.out.println("MaximumContiguousSubsequenceSum is "+max); 
59           System.out.println("Relevant Subsequence is " + sol); 
60       }
