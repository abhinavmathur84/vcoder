14       //Approach 
15    
16       /** 
17        * lp(i,j) = length of longest palindromic subsequence between a[i] and a[j] 
18        * sols(i,j) = longest palindromic subsequence between a[i] and a[j] 
19        * lp(i,j)  = 1 if i=j 
20        *          = 1 if (j=i+1 and a[i] != a[j]) 
21        *          = 2 if (j=i+1 and a[i] = a[j]) 
22        *          = lp(i+1,j-1) +2 if a(i) = a(j) 
23        *          = max {lp(i+1,j),lp(i,j-1)} 
24        * <p/> 
25        * We need to calculate lp(0,a.length-1) 
26        */ 
27       void calculateLPS() { 
28           lp = new int[a.length][a.length]; 
29           sols = new String[a.length][a.length]; 
30    
31           for (int i = 0; i < a.length; i++) { // solution for substring of length 1 
32               lp[i][i] = 1; 
33               sols[i][i] = a[i]; 
34           } 
35           // build the table for all substring length >=2 till a.length 
36           int subStrLen = 2; 
37           for (subStrLen = 2; subStrLen <= a.length; subStrLen++) { 
38               // when substring length is subStrLen, i can take value from 0 to a.length-subStrLen+1 
39               for (int i = 0; i < a.length - subStrLen + 1; i++) { 
40                   // corresponding j will be from i+subStrLen -1 
41                   int j = i + subStrLen - 1; 
42                   if (a[i].equals(a[j]) && subStrLen == 2) { // if substring of length 2 and both characters are same 
43                       lp[i][j] = 2; // palindrome length will be 2 
44                       sols[i][j] = a[i] + a[j]; // palindrome will be the substring itself 
45                   } else if (a[i].equals(a[j])) { // if both characters are same 
46                       lp[i][j] = lp[i + 1][j - 1] + 2; // palindrome length is sub problem + start and end character 
47                       sols[i][j] = a[i] + sols[i + 1][j - 1] + a[j]; 
48                   } else { 
49                       int t1 = lp[i][j - 1]; // else solve these 2 sub problems 
50                       int t2 = lp[i + 1][j]; 
51                       if (t1 > t2) { // pick the larger one 
52                           lp[i][j] = t1; 
53                           sols[i][j] = sols[i][j - 1]; 
54                       } else { 
55                           lp[i][j] = t2; 
56                           sols[i][j] = sols[i + 1][j]; 
57                       } 
58                   } 
59               } 
60           } 
61           System.out.println("Length of longest palindromic subsequence " + lp[0][a.length - 1]); 
62           System.out.println("Longest palindromic subsequence is " + sols[0][a.length - 1]); 
63       }
