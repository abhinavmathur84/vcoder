4    /** 
5     * Given an array p[] which represents the chain of matrices 
6     * such that the ith matrix Ai is of dimension p[i-1] x p[i]. 
7     * We need to write a function MatrixChainOrder() that should 
8     * return the minimum number of multiplications needed to multiply 
9     * the chain. 
10    */ 
11   public class MatrixChainMultiplication { 
12    
13       //we have A1....An 
14       //dimension of Ai is p[i-1]*p[i] 
15       // dimension of A1...An-1   p[0]*p[n-1] 
16       int[] p; 
17       String[] matrix; 
18    
19       int[][] M; 
20       String[][] sols; 
21    
22       //Approach 
23    
24       /** 
25        * M[i][j] = minimum number of multiplications required to multiply Ai...Aj 
26        * T = Ai......Aj of dimension p[i-1]*p[j] 
27        * T1 = Ai...Ak of dimension p[i-1]*p[k] 
28        * T2 = Ak+1...Aj of dimension p[k]*p[j] 
29        * so number of multiplications required for T 
30        * M[i][j] = min {M[i][k]+ M[k+1][j] + p[i-1]*p[k]*p[j]} 
31        * We need to calculate M[1][n-1] 
32        * M[i][j] = 0 when i=j as no multiplications are required 
33        * Number of matrix is p.length-1. 
34        */ 
35    
36       void calculateMatMult() { 
37           M = new int[p.length + 1][p.length + 1]; 
38           sols = new String[p.length + 1][p.length + 1]; 
39           int n = p.length; 
40           for (int i = 1; i < n; i++) { 
41               M[i][i] = 0; 
42               sols[i][i] = matrix[i - 1]; 
43           } 
44           for (int len = 2; len <= n; len++) { 
45               // len = 2 A1A2 i=1.n-1, j=2 if k=i(A1)(A2) and k<j 
46               // len = n A1...An, i=1 j=n k<j 
47               // for any other len i=n-len+1; j= i+len-1 k<j 
48               for (int i = 1; i < n - len + 1; i++) { 
49                   int j = i + len - 1; 
50                   int min = Integer.MAX_VALUE; 
51                   String sol = ""; 
52                   for (int k = i; k < j; k++) { 
53                       int t = M[i][k] + M[k + 1][j] + p[i - 1] * p[k] * p[j]; //(Ai...Ak)(Ak+1..Aj)=> p[i-1]*p[k]*p[j-1] 
54                       if (t < min) { 
55                           min = t; 
56                           int temp = k + 1; 
57                           sol = "(" + sols[i][k] + ")(" + sols[k + 1][j] + ")"; 
58                       } 
59                   } 
60                   M[i][j] = min; 
61                   sols[i][j] = sol; 
62               } 
63           } 
64           System.out.println(M[1][n - 1]); 
65           System.out.println(sols[1][n - 1]); 
66       }

68       public static void main(String[] args) { 
69           int p[] = {40, 20, 30, 10, 30}; 
70           String[] mat = new String[]{"A", "B", "C", "D"}; 
71           MatrixChainMultiplication mcm = new MatrixChainMultiplication(); 
72           mcm.p = p; 
73           mcm.matrix = mat; 
74           mcm.calculateMatMult();
