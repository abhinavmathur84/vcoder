4     
5        //priceLength[i] = price of piece of length i 
6        int[] priceLength; 
7        int length;

9        int[] cost;

13    
14       /** 
15        * C[i] = Optimal cost to cut a cost of length i; 
16        * C[i] = max{C[i-l]+priceLength[l]} for all l<i 
17        */ 
18       void calculateCost() { 
19    
20           cost = new int[length + 1]; 
21           sols = new String[length + 1]; 
22           // initialize the optimal solution to be the price of piece of that length itself 
23           for (int i = 1; i <= length; i++) { 
24               cost[i] = priceLength[i - 1]; 
25               sols[i] = i + ""; 
26           } 
27           cost[0] = 0; 
28           sols[0] = ""; 
29           // build the solution bottom up 
30           for (int i = 1; i <= length; i++) { 
31               int max = Integer.MIN_VALUE; 
32               for (int j = 1; j <= i; j++) { 
33                   int t = cost[i - j] + priceLength[j]; // check for all pieces of length less than i 
34                   if (t > max) { 
35                       max = t; 
36                       cost[i] = t; 
37                       sols[i] = sols[i - j] + "," + j; // include piece of length j and remaining subproblem 
38                   } 
39               } 
40           } 
41           System.out.println("Optimal Cost is " + cost[length]); 
42           System.out.println("Optimal Solution is " + sols[length]); 
43       }
